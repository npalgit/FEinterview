    /**88. Merge Sorted Array 
     * @param nums1
     * @param m
     * @param nums2
     * @param n
     * 两个有序数组，合并
     */
    public void merge(int[] nums1, int m, int[] nums2, int n) {
    	int x=m-1, y=n-1;
    	for (int i=m+n-1; i>=0; i--) {
    		if (x>=0 && y>=0) {
    			if (nums1[x] > nums2[y]) {
        			nums1[i] = nums1[x];
        			x--;
        		} else{
        			nums1[i] = nums2[y];
        			y--;
        		}
    		} else if (x>=0) {
    			nums1[i] = nums1[x];
    			x--;
    		} else {
    			nums1[i] = nums2[y];
    			y--;
    		}
    		
    	}
    }
