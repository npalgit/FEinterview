	/**2. Add Two Numbers
	 * @param l1
	 * @param l2
	 * @return 逆序的两个数相加，保存个位数
	 */
	public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
		ListNode l3 = new ListNode(-1);
		ListNode head = l3;
		int sum = 0;
		while (l1 != null && l2 != null) {
			sum += l1.val + l2.val;
			ListNode cur = new ListNode(sum%10);
			l3.next = cur;
			sum = sum/10;
			l1 = l1.next;
			l2 = l2.next;
			l3 = l3.next;
		}
		while (l1 != null) {
			sum += l1.val;
			ListNode cur = new ListNode(sum%10);
			l3.next = cur;
			sum = sum/10;
			l1 = l1.next;
			l3 = l3.next;
		}
		while (l2 != null) {
			sum += l2.val;
			ListNode cur = new ListNode(sum%10);
			l3.next = cur;
			sum = sum/10;
			l2 = l2.next;
			l3 = l3.next;
		}
		if (sum != 0) {
			ListNode cur = new ListNode(sum);
			l3.next = cur;
		}
		return head.next;
	}
