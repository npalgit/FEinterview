```
最大递增子序列为 1 5 6 8 9 

dp[i]是以i结尾的最大递增子序列长度
dp[i] = max( dp[j] ) +1  , j<i && a[j]<a[i]
开始：dp[i] = 1

```

```js
var dp = [1, 1, 1, 1]
for (let i = 0; i < n; i++) {
  for (let j = 0; j < i; j++) {
    if (a[j] < a[i]) {
      dp[i] = max(dp[i], dp[j]+1)
    }
  }
}
var ans = 0
for (let i in dp) {
  ans = max(ans, dp[i])
}
```
