	/**83. Remove Duplicates from Sorted List
	 * @param head
	 * @return 链表去重
	 */
	public ListNode deleteDuplicates(ListNode head) {
		if (head == null) {
			return null;
		}
		ListNode ret = new ListNode(-1);
		ret.next = head;
		ListNode pre = head;
		head = head.next;
		while(head != null) {
			if (head.val == pre.val) {
				pre.next = head.next;
			}else {
				pre = head;
			}
			head = head.next;
		}
		return ret.next;
	}
